#Install Apache.
	sudo apt update
	sudo apt install apache2

#Adjust firewall
	sudo ufw app list

#Leave Apache Restricted 
	sudo ufw allow 'Apache'

#Verify change
	sudo ufw status

#If it shows the firewall as inactive
	sudo ufw enable

#Checking web server
	sudo systemctl status apache2
